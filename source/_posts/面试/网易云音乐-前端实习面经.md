6. 写一个判断对象是否存在循环引用的函数

   ```javascript
   const s = [];
   function fun(obj){
     if(typeof obj !== 'object'){
       return;
     }
     s.push(obj);
     let flag = false;
     Object.keys(obj).forEach(key => {
       if(typeof obj[key] === 'object') {
         if(s.includes(obj[key])){
           flag = true;
         } else{
           flag = fun(obj[key]);
         }
       }
     });
      s.pop();
     return flag;
   }
   ```
